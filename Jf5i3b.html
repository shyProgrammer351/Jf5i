코드 빈칸 채우기

import javax.swing.JOptionPane; 
public class EvenOdd_Main 
{ 
	public static void main (String[] args) 
	{ 
		String numStr, result; 
		int num, again; 
 
		do  
		{ 
			numStr = JOptionPane. ("Enter an integer: ");	 
			num = (numStr); 	 
			result = "That number is " + ((num%2 == 0) ? "even" : "odd"); 
			JOptionPane. (null, result);	 
			again = JOptionPane. (null, "Do Another?");	 
		} while( again == JOptionPane. );	 
		return; 
	} 
}
