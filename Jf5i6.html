문제

On을 누르면 전구가 켜지고 Off를 누르면 전구가 꺼지는 GUI 문제입니다 이를 해결하는 다음 프로그램을 해석하세요

알고리즘

 프로그램 시작 
    큰 패널에 전구 이미지 패널과 전구 제어 패널 두 개를 추가 
    프레임에 큰 패널을 추가하고 화면에 출력 
 프로그램 종료 
 
 전구 이미지 패널 준비 
    전구가 켜진 이미지와 전구가 꺼진 이미지 준비 
    전구가 켜진 이미지로 초기화하여 출력 
 
 전구 제어 패널 준비 
    on 버튼 및 리스너 등록 
    off 버튼 및 리스너 등록 
 
 on 버튼 클릭시 작동 
    전구를 켜고 on 버튼 비활성화 
    나중에 전구를 끌 수 있도록 off 버튼 활성화 
 
 off 버튼 클릭시 작동 시작 
    전구를 끄고 off 버튼 비활성화 
    나중에 전구를 켤 수 있도록 on 버튼 활성화

프로그램 코드 

	// 파일명 : ./Chapter05/LightBulb_Main.java 

	import javax.swing.*; 

	import java.awt.*; 

	import java.awt.event.*; 

 

	class LightBulbControls extends JPanel	 

	{ 

		private LightBulbPanel bulb; 

		private JButton onButton, offButton; 

 

		// 전구 제어 패널 준비 

1		public LightBulbControls (LightBulbPanel bulbPanel) 

		{ 

			// 전구 이미지를 변경할 수 있도록 준비  

			bulb = bulbPanel; 

 

			// on 버튼 및 리스너 등록 

			onButton = new JButton ("On"); 

			onButton.setEnabled (false); 

			onButton.setMnemonic ('n'); 

			onButton.setToolTipText ("Turn it on!"); 

			onButton.addActionListener (new OnListener());	 

 

			// off 버튼 및 리스너 등록 

			offButton = new JButton ("Off"); 

			offButton.setEnabled (true); 

			offButton.setMnemonic ('f'); 

			offButton.setToolTipText ("Turn it off!"); 

			offButton.addActionListener (new OffListener());	 

 

			setBackground (Color.black); 

			add (onButton); 

			add (offButton); 

15		} 

		private class OnListener implements ActionListener	 

		{ 

			// on 버튼 클릭시 작동 

16			public void actionPerformed (ActionEvent event)	 

			{ 

				// 전구를 켜고 on 버튼 비활성화 

17				bulb.setOn (true); 

18				onButton.setEnabled (false); 

				// 나중에 전구를 끌 수 있도록 off 버튼 활성화 

19				offButton.setEnabled (true); 

				// 전구 이미지 패널 화면 업데이트  

20				bulb.repaint(); 

			} 

		} 

 

		private class OffListener implements ActionListener	 

		{ 

			// off 버튼 클릭시 작동 시작 

21			public void actionPerformed (ActionEvent event)	 

			{ 

				// 전구를 끄고 off 버튼 비활성화 

22				bulb.setOn (false); 

				// 나중에 전구를 켤 수 있도록 on 버튼 활성화 

				onButton.setEnabled (true); 

				offButton.setEnabled (false); 

				// 전구 이미지 패널 화면 업데이트   

25				bulb.repaint(); 

			} 

		} 

	} 

 

	class LightBulbPanel extends JPanel	 

	{ 

		private boolean on; 

		private ImageIcon lightOn, lightOff; 

		private JLabel imageLabel; 

 

		// 전구 이미지 패널 준비 

26		public LightBulbPanel() 

		{ 

 

			// 전구가 켜진 이미지와 전구가 꺼진 이미지 준비  

			lightOn = new ImageIcon ("./Chapter05/image/lightBulbOn.gif"); 

			lightOff = new ImageIcon ("./Chapter05/image/lightBulbOff.gif"); 

 

			setBackground (Color.black); 

 

			// 전구가 켜진 이미지로 초기화하여 출력 

			on = true; 

			imageLabel = new JLabel (lightOff); 

			add (imageLabel); 

32		} 

 

33		public void paintComponent (Graphics page) 

		{ 

			super.paintComponent(page); 

 

			if (on) 

				imageLabel.setIcon (lightOn); 

			else 

				imageLabel.setIcon (lightOff); 

38		} 

 

39		public void setOn (boolean lightBulbOn) 

		{ 

40			on = lightBulbOn; 

		} 

	} 

 

58	// off 버튼을 클릭 

59	// 프레임 우측상단 X버튼 클릭하기(종료)	 

	public class LightBulb_Main 

	{ 

		// 프로그램 시작 

41		public static void main (String[] args) 

		{ 

			// 프레임 생성  

			JFrame frame = new JFrame ("Light Bulb"); 

			frame.setDefaultCloseOperation (JFrame.EXIT_ON_CLOSE); 

			// 전구 이미지 패널과 전구 제어 패널 생성  

44			LightBulbPanel bulb 

441			                   = new LightBulbPanel(); 

45			LightBulbControls controls 

451			                   = new LightBulbControls (bulb); 

			// 큰 패널에 전구 이미지 패널과 전구 제어 패널 두 개를 추가 

			JPanel panel = new JPanel(); 

			panel.setBackground (Color.black); 

			panel.setLayout (new BoxLayout(panel, BoxLayout.Y_AXIS)); 

			panel.add (Box.createRigidArea (new Dimension (0, 20))); 

			panel.add (bulb); 

			panel.add (Box.createRigidArea (new Dimension (0, 10))); 

			panel.add (controls); 

			panel.add (Box.createRigidArea (new Dimension (0, 10))); 

 

			// 프레임에 큰 패널을 추가하고 화면에 출력 

			frame.getContentPane().add( panel );		 

			frame.pack(); 

56			frame.setVisible(true); 

			// 프로그램 종료 

57			return; 

		} 

	}
