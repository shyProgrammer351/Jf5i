문제

Say it with style!를 진하게 또는 이탤릭체로 출력하는 문제입니다 이를 해결하는 다음 프로그램에 대해 빈칸을 채우세요

코드 빈칸 채우기

import javax.swing.JFrame; 
import javax.swing.*; 
import java.awt.*; 
import java.awt.event.*; 
 
class StyleOptionsPanel  JPanel   
{ 
	private JLabel saying; 
	private JCheckBox bold, italic; 
 
	public StyleOptionsPanel() 
	{ 
		saying = new JLabel( "Say it with style!" ); 
		saying.( new Font("Helvetica", Font.PLAIN, 36 ) ); 
 
		bold = new JCheckBox ("Bold"); 
		bold.setBackground (Color.cyan); 
		 
		italic = new JCheckBox ("Italic"); 
		italic.setBackground (Color.cyan); 
 
		StyleListener listener = new StyleListener(); 
		bold.(listener);	 
		italic.(listener);	 
 
		add (saying); 
		add (bold); 
		add (italic); 
 
		setBackground (Color.cyan); 
		setPreferredSize (new Dimension(300, 100)); 
	} 
	private class StyleListener  ItemListener	 
	{ 
		public void (ItemEvent event)	 
		{ 
			int style = Font.PLAIN; 
 
			if (bold.isSelected()) 
				style = Font.BOLD; 
 
			if (italic.isSelected()) 
				style += Font.ITALIC; 
 
			saying. (new Font ("Helvetica", style, 36));	 
			return; 
		} 
	} 
} 
 
public class StyleOptions_Main 
{ 
	public static void main (String[] args) 
	{ 
		JFrame frame = new JFrame ("Style Options"); 
		frame.setDefaultCloseOperation (JFrame.EXIT_ON_CLOSE); 
 
		frame.getContentPane().add ( 
						new StyleOptionsPanel()); 
 
		frame.pack(); 
		frame.setVisible(true); 
		 
		return; 
	} 
}
