문제

숫자를 입력받아 홀수인지 짝수인지 맞추는 문제입니다 이를 해결하는 다음 프로그램을 해석하세요

알고리즘

프로그램시작 
   숫자 입력 
   홀수인지 짝수인지 출력 
   계속 반복할지 확인 
프로그램 종료

프로그램 코드 

	// 파일명 : ./Chapter05/EvenOdd_Main.java 

	import javax.swing.JOptionPane; 

 

	public class EvenOdd_Main 

	{ 

		// 프로그램시작 

1		public static void main (String[] args) 

		{ 

			String numStr, result; 

			int num, again; 

	 

2			do  

			{ 

				// 숫자 입력 

				numStr = JOptionPane.showInputDialog ("Enter an integer: ");	 

				// 35, 7의 순서대로 입력한다고 가정 

3				num = Integer.parseInt(numStr); 	 

 

				// 홀수인지 짝수인지 출력 

				result = "That number is " + ((num%2 == 0) ? "even" : "odd"); 

4				JOptionPane.showMessageDialog (null, result);	 

 

				// 계속 반복할지 확인 

				// 예, 아니오의 순서대로 선택한다고 가정  

5				again = JOptionPane.showConfirmDialog (null, "Do Another?");	 

6			} while( again == JOptionPane.YES_OPTION );	 

 

			// 프로그램 종료 

7			return; 

		} 

	}
