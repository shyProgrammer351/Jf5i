프로그램 코드 

	// 파일명 : ./Chapter06/PushCounter_Main.java 

	import javax.swing.JFrame; 

	import java.awt.*; 

	import java.awt.event.*; 

	import javax.swing.*; 

	 

	class PushCounterPanel extends JPanel	 

	{ 

		private int count; 

		private JButton push; 

		private JLabel label; 

	 

		// 패널 준비 

1		public PushCounterPanel () 		 

		{ 

			count = 0; 

	 

			// Push Me! 버튼 준비 

			push = new JButton ("Push Me!"); 

2			push.addActionListener (new ButtonListener());	 

	 

			// Pushes 횟수 출력 준비 

3			label = new JLabel ("Pushes: " + count); 

	 

			add (push); 

4			add (label); 

	 

			setBackground (Color.cyan); 

5			setPreferredSize (new Dimension(300, 40)); 

		} 

	 

		private class ButtonListener implements ActionListener 	 

		{ 

			// 버튼 클릭시 작동 

6			public void actionPerformed (ActionEvent event) 	 

			{ 

				// 클릭 횟수 증가 후 출력 

				count++; 

7				label.setText("Pushes: " + count);	 

8				return; 

			} 

		} 

	} 

	 

9	// Push Me! 버튼 첫번째 클릭하기

	public class PushCounter_Main 

	{ 

		// 프로그램시작 

12		public static void main (String[] args) 

		{ 

			// 틀 준비 

			JFrame frame = new JFrame ("Push Counter"); 

13			frame.setDefaultCloseOperation (JFrame.EXIT_ON_CLOSE); 

	 

			// 패널 준비 

14			PushCounterPanel panel =  

15						new PushCounterPanel(); 

16			frame.getContentPane().add(panel); 

	 

			// 틀에 패널 끼우고 실행 준비 완료 

			frame.pack(); 

17			frame.setVisible(true); 

			 

			// 프로그램 종료 

18			return; 

		} 

	}
