문제

On을 누르면 전구가 켜지고 Off를 누르면 전구가 꺼지는 GUI 문제입니다 이를 해결하는 다음 프로그램에 대해 빈칸을 채우세요

코드 빈칸 채우기

 javax.swing.*; 
 java.awt.*; 
 java.awt.event.*; 
 LightBulbControls  JPanel	 
{ 
	private LightBulbPanel bulb; 
	private JButton onButton, offButton; 
	 LightBulbControls( LightBulbPanel bulbPanel ) 
	{ 
		bulb = bulbPanel; 
		onButton =  JButton ("On"); 
		onButton.setEnabled (false); 
		onButton.setMnemonic ('n'); 
		onButton.setToolTipText ("Turn it on!"); 
		onButton. (new OnListener());	 
		offButton =  JButton ("Off"); 
		offButton.setEnabled (true); 
		offButton.setMnemonic ('f'); 
		offButton.setToolTipText ("Turn it off!"); 
		offButton. (new OffListener());	 
		setBackground (Color.black); 
		add (onButton); 
		add (offButton); 
	} 
	private  OnListener  	 
	{ 
		 void  (ActionEvent event)	 
		{ 
			bulb.setOn (true); 
			onButton.setEnabled (false); 
			offButton.setEnabled (true); 
			bulb.repaint(); 
		} 
	} 
	private  OffListener  	 
	{ 
		 void  (ActionEvent event)	 
		{ 
			bulb.setOn (false); 
			onButton.setEnabled (true); 
			offButton.setEnabled (false); 
			bulb.repaint(); 
		} 
	} 
} 
 LightBulbPanel  JPanel	 
{ 
	private boolean on; 
	private ImageIcon lightOn, lightOff; 
	private JLabel imageLabel; 
	 LightBulbPanel() 
	{ 
		lightOn =  ImageIcon ("./Chapter05/image/lightBulbOn.gif"); 
		lightOff =  ImageIcon ("./Chapter05/image/lightBulbOff.gif"); 
		setBackground (Color.black); 
		on = true; 
		imageLabel =  JLabel (lightOff); 
		add (imageLabel); 
	} 
	 void paintComponent (Graphics page) 
	{ 
		super.paintComponent(page); 
		if (on) 
			imageLabel.setIcon (lightOn); 
		else 
			imageLabel.setIcon (lightOff); 
	} 
	 void setOn (boolean lightBulbOn) 
	{ 
		on = lightBulbOn; 
	} 
} 
  LightBulb_Main 
{ 
	 static void main (String[] args) 
	{ 
		JFrame frame =  JFrame ("Light Bulb"); 
		frame.setDefaultCloseOperation (JFrame.EXIT_ON_CLOSE); 
		LightBulbPanel bulb 
		                   =  LightBulbPanel(); 
		LightBulbControls controls 
		                   =  LightBulbControls (bulb); 
		JPanel panel =  JPanel(); 
		panel.setBackground (Color.black); 
		panel.setLayout (new BoxLayout(panel, BoxLayout.Y_AXIS)); 
		panel.add (Box.createRigidArea (new Dimension (0, 20))); 
		panel.add (bulb); 
		panel.add (Box.createRigidArea (new Dimension (0, 10))); 
		panel.add (controls); 
		panel.add (Box.createRigidArea (new Dimension (0, 10))); 
		frame.getContentPane().add( panel );		 
		frame.pack(); 
		frame.setVisible(true); 
		return; 
	} 
}
