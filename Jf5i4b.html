문제

웃는 모양의 아이콘이 틀 안에서 상하좌우로 계속 움직이는 GUI 문제입니다 이를 해결하는 다음 프로그램에 대해 빈칸을 채우세요

코드 빈칸 채우기

 java.awt.*; 
 java.awt.event.*; 
 javax.swing.*; 
 ReboundPanel  JPanel	 
{ 
	private final int WIDTH = 300, HEIGHT = 100; 
	private final int DELAY = 20, IMAGE_SIZE = 35; 
	private ImageIcon image; 
	private Timer timer; 
	private int x, y, moveX, moveY; 
	 
	 ReboundPanel() 
	{ 
		timer =  Timer(DELAY,  ReboundListener()); 
		image =  ImageIcon ("./Chapter05/image/happyFace.gif"); 
		x = 0; 
		y = 40; 
		moveX = moveY = 3; 
		setPreferredSize (new Dimension(WIDTH, HEIGHT)); 
		setBackground (Color.black); 
		timer.start(); 
	} 
	 void paintComponent (Graphics page) 
	{ 
		super.paintComponent (page); 
		image.paintIcon (this, page, x, y); 
	} 
	private  ReboundListener  	 
	{ 
		 void  (ActionEvent event)	 
		{ 
			x += moveX; 
			y += moveY; 
			if (x <= 0 || x >= WIDTH-IMAGE_SIZE) 
				moveX = moveX * -1; 
			 
			if (y <= 0 || y >= HEIGHT-IMAGE_SIZE) 
				moveY = moveY * -1; 
			 
			();	 
		} 
	} 
} 
  Rebound_Main 
{ 
	 static void main (String[] args) 
	{ 
		JFrame frame =  JFrame ("Rebound"); 
		frame.setDefaultCloseOperation (JFrame.EXIT_ON_CLOSE); 
		frame.getContentPane().add( 
								 ReboundPanel()); 
		frame.pack(); 
		frame.setVisible(true); 
		return; 
	} 
}
