      / 
  [   옷   문
      -> 
WASD와 [Enter]를 입력하세요: a
      / 
 옷]       문

import java.util.Scanner; 
	 
public class Direction 
{ 
	public static void main( String[] args ) { 
		Scanner scan = new Scanner( System.in ); 
		 direction = '\0'; 
	 
		System.out.println( "      / " ); 
		System.out.println( "  [   옷   문" ); 
		System.out.println( "      -> \n" ); 
		 
		System.out.print( "방향키(WASD)와 [Enter]를 입력하세요: " ); 
		direction= scan.nextLine().charAt(0); 
	 
		System.out.print( " [2;6f  " ); 
		( direction ) { 
			 'w'  'W'  
				System.out.print( " [1;6f 옷/ [7;15f" ); 
				; 
			 'a'  'A'  
				System.out.print( " [2;1f 옷] [7;15f" ); 
				; 
			 's'  'S'  
				System.out.print( " [3;6f 옷-> [7;15f" ); 
				; 
			 'd'  'D'  
				System.out.print( " [2J [4;15fSUCCESS! [7;15f" ); 
		} 
	 
	} 
}
