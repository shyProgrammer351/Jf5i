문제

마우스로 클릭하면 점이 생기고 해당 좌표를 알려주는 GUI 문제입니다 이를 해결하는 다음 프로그램에 대해 빈칸을 채우세요

코드 빈칸 채우기

 javax.swing.JFrame; 
 javax.swing.JPanel; 
 java.awt.*; 
 java.awt.event.*; 
 
 CoordinatesPanel  JPanel	 
{ 
	private final int SIZE = 6; 
	private int x = 50, y = 50;  
 
	 CoordinatesPanel() 
	{ 
		(  CoordinatesListener() );	 
		setBackground( Color.black ); 
		setPreferredSize(  Dimension(300, 200) ); 
	} 
	 void paintComponent (Graphics page) 
	{ 
		super.paintComponent(page); 
		page.setColor (Color.green); 
		page.fillOval (x, y, SIZE, SIZE); 
		page.drawString ("Coordinates: (" + x + ", " + y + ")", 5, 15); 
	} 
	private  CoordinatesListener  	 
	{ 
		 void  (MouseEvent event)	 
		{ 
			x = event.getX(); 
			y = event.getY(); 
			repaint(); 
		} 
		 void mouseClicked (MouseEvent event) {} 
		 void mouseReleased (MouseEvent event) {} 
		 void mouseEntered (MouseEvent event) {} 
		 void mouseExited (MouseEvent event) {} 
	} 
} 
 
 
  Coordinates_Main 
{ 
	 static void main (String[] args) 
	{ 
		JFrame frame =  JFrame ("Coordinates"); 
		frame.setDefaultCloseOperation (JFrame.EXIT_ON_CLOSE); 
		frame.getContentPane().add ( 
				 CoordinatesPanel()); 
		frame.pack(); 
		frame.setVisible(true); 
		return; 
	} 
}
