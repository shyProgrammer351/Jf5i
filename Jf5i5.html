문제

마우스로 클릭하면 점이 생기고 해당 좌표를 알려주는 GUI 문제입니다 이를 해결하는 다음 프로그램을 해석하세요

알고리즘

 프로그램시작 
    프레임 준비 
    패널 생성하고 후 프레임에 추가 
    프레임에 묶어준 뒤 화면에 출력 
 프로그램 종료 
 
 패널 준비 
    패널 내에서 마우스 이벤트 발생시 반응하도록 준비 
 
 마우스 이벤트 발생시 작동 
    현재 마우스의 xy 좌표 값 확인 
    패널 화면 업데이트

프로그램 코드 

	// 파일명 : ./Chapter05/Coordinates_Main.java 

	//다른 좌표를 한번 찍고 종료하는 실행 순서 

	import javax.swing.JFrame; 

	import javax.swing.JPanel; 

	import java.awt.*; 

	import java.awt.event.*; 

	 

	class CoordinatesPanel extends JPanel	 

	{ 

		private final int SIZE = 6; 

		private int x = 50, y = 50;  

	 

		// 패널 준비 

3		public CoordinatesPanel() 

		{ 

			// 패널 내에서 마우스 이벤트 발생시 반응하도록 준비 

			addMouseListener( new CoordinatesListener() );	 

			setBackground( Color.black ); 

6			setPreferredSize( new Dimension(300, 200) ); 

		} 

 

7		public void paintComponent (Graphics page) 

		{ 

			super.paintComponent(page); 

 

			page.setColor (Color.green); 

 

			page.fillOval (x, y, SIZE, SIZE); 

 

11			page.drawString ("Coordinates: (" + x + ", " + y + ")", 5, 15); 

		} 

 

		private class CoordinatesListener implements MouseListener	 

		{ 

			// 마우스 이벤트 발생시 작동  

12			public void mousePressed (MouseEvent event)	 

			{ 

				// 현재 마우스의 xy 좌표 값 확인   

				x = event.getX(); 

				y = event.getY(); 

				// 패널 화면 업데이트 

15				repaint(); 

			} 

 

			public void mouseClicked (MouseEvent event) {} 

			public void mouseReleased (MouseEvent event) {} 

			public void mouseEntered (MouseEvent event) {} 

19			public void mouseExited (MouseEvent event) {} 

		} 

	} 

	 

28	// (211,142) 좌표에 클릭 

29	// 프레임 우측상단 X버튼 클릭하기(종료)	 

	 

	public class Coordinates_Main 

	{ 

		// 프로그램시작 

20		public static void main (String[] args) 

		{ 

			// 프레임 준비 

			JFrame frame = new JFrame ("Coordinates"); 

			frame.setDefaultCloseOperation (JFrame.EXIT_ON_CLOSE); 

			// 패널 생성하고 후 프레임에 추가 

23			frame.getContentPane().add ( 

24					new CoordinatesPanel()); 

			// 프레임에 묶어준 뒤 화면에 출력 

			frame.pack(); 

26			frame.setVisible(true); 

			// 프로그램 종료 

27			return; 

		} 

	}
