알고리즘

프로그램 시작 
   캐릭터의 초기 위치를 출력 
   이동 방향 입력 받기 
   이동 결과 출력 
      W는 위로 이동 
      A는 왼쪽으로 이동 
      S는 아래로 이동 
      D는 오른쪽으로 이동 
프로그램 종료

프로그램 코드 

	// 파일명 : ./Chapter05/Direction.java

	import java.util.Scanner;

		 

	public class Direction

	{

		// 프로그램 시작 

1		public static void main( String[] args ) {

			Scanner scan = new Scanner( System.in );

2			char direction = '\0';

3			System.out.println( "      / " );

4			System.out.println( "  [   옷   문" );

5			System.out.println( "      -> \n" );

			 

			// 이동 방향 입력 받기 

6			System.out.print( "방향키(WASD)와 [Enter]를 입력하세요: " );

7			direction= scan.nextLine().charAt(0);

		 

			// 이동 결과 출력 

8			System.out.print( " \033[2;6f  " );

9			switch( direction ) {

				// W는 위로 이동 

10				case 'w': case 'W':

11					System.out.print( " \033[1;6f 옷/ \033[7;15f" );

12					break;

				// A는 왼쪽으로 이동 

13				case 'a': case 'A':

14					System.out.print( " \033[2;1f 옷] \033[7;15f" );

15					break;

				// S는 아래로 이동 

16				case 's': case 'S':

17					System.out.print( " \033[3;6f 옷-> \033[7;15f" );

18					break;

				// D는 오른쪽으로 이동 

19				case 'd': case 'D':

20					System.out.print( " \033[2J \033[4;15fSUCCESS! \033[7;15f" );

			}

		 

		// 프로그램 종료 

21		}

	}
