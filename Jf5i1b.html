문제

Push Me! 버튼을 누르면 Pushes 횟수가 증가하는 문제입니다 이를 해결하는 다음 프로그램에 대해 빈칸을 채우세요

코드 빈칸 채우기

 javax.swing.JFrame; 
 java.awt.*; 
 java.awt.event.*; 
 javax.swing.*; 
 
 PushCounterPanel  JPanel	 
{ 
	private int count; 
	private JButton push; 
	private JLabel label; 
 
	 PushCounterPanel () 		 
	{ 
		count = 0; 
 
		push =  JButton ("Push Me!"); 
		push. (new ButtonListener());	 
 
		label =  JLabel ("Pushes: " + count); 
 
		add (push); 
		add (label); 
 
		setBackground (Color.cyan); 
		setPreferredSize (new Dimension(300, 40)); 
	} 
 
	private  ButtonListener  ActionListener 	 
	{ 
		 void  (ActionEvent event) 	 
		{ 
			count++; 
			label.("Pushes: " + count);	 
			return; 
		} 
	} 
} 
 
  PushCounter_Main 
{ 
	 static void main (String[] args) 
	{ 
		JFrame frame =  JFrame ("Push Counter"); 
		frame.setDefaultCloseOperation (JFrame.EXIT_ON_CLOSE); 
 
		PushCounterPanel panel =  
					 PushCounterPanel(); 
		frame.getContentPane().add(panel); 
 
		frame.pack(); 
		frame.setVisible(true); 
		 
		return; 
	}
